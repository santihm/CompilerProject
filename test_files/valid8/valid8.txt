program

saldo : int;

def depositar(monto : int) {
    saldo = saldo + monto;
}

def retirar(monto : int) {
    if (monto > saldo) then {
        "Saldo insuficiente!" !>> ;
    } else {
        saldo = saldo - monto;
    }
}

def consultar() {
    "Saldo actual:", saldo !>> ;
}

    "Saldo inicial:" !>> ;
