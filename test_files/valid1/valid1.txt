// Programma esemplificativo in sintassi Toy3

program

// Dichiarazione variabili globali
c: 1;
a | b | x : double;
taglia | ans1 | ans : string;
risultato: double;

// Definizione delle funzioni
def sommac(a: double; d: double; b: double; ref size: string; ref result: double)
{
    result := a + b + c + d;

    if (result > 100) then {
        size := "grande";
    }
    else {
        if (result > 50) then {
            size := "media";
        }
        else {
            size := "piccola";
        }
    }
}

def stampa(messaggio: string): string
{
    i: 0;
    while (i < 4) do {
        "" !>>; // Ritorno a capo
        i := i + 1;
    }
    messaggio !>>;
    return "ok";
}

begin
    a := 1;
    b := 2.2;
    x := 3;
    risultato := 0.0;
    ans := "no";

    sommac(a, x, b, taglia, risultato);

    stampa("La somma di " + a + " e " + b + " incrementata di " + c + " è " + taglia);
    stampa("Ed è pari a " + risultato);

    "Vuoi continuare? (si/no) - inserisci due volte la risposta\n">>;
    ans <<;
    ans1 <<;

    while (ans == "si") do {
        "Inserisci un intero: ">>;
        a <<;
        "Inserisci un reale: ">>;
        b <<;

        sommac(a, x, b, taglia, risultato);

        stampa("La somma di " + a + " e " + b + " incrementata di " + c + " è " + taglia);
        stampa("Ed è pari a " + risultato);

        "Vuoi continuare? (si/no): ">>;
        ans <<;
    }

    "" !>>; // Ritorno a capo finale
    "Ciao" !>>;
end
